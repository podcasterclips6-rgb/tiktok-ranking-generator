Monitor Progress with Polling


const pollStatus = async (jobId) => {
  return new Promise((resolve, reject) => {
    const interval = setInterval(async () => {
      try {
        const response = await fetch(
          `http://localhost:5000/api/video/status/${jobId}`
        );
        const data = await response.json();

        console.log(`Progress: ${data.progress}% - ${data.message}`);

        if (data.status === 'completed') {
          clearInterval(interval);
          resolve(data);
        } else if (data.status === 'error') {
          clearInterval(interval);
          reject(new Error(data.message));
        }
      } catch (error) {
        clearInterval(interval);
        reject(error);
      }
    }, 1000);
  });
};

try {
  const result = await pollStatus(jobId);
  console.log('Video ready at:', result.videoUrl);
} catch (error) {
  console.error('Generation failed:', error);
}

const generateAndDownloadVideo = async (tiktokLinks) => {
  try {

    const genResponse = await fetch('http://localhost:5000/api/video/generate', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({

      })
    });

    const { jobId } = await genResponse.json();

    // Step 2: Monitor progress
    console.log('Waiting for completion...');
    const statusResponse = await fetch(
      `http://localhost:5000/api/video/status/${jobId}`
    );

      throw new Error('Video generation failed: ' + status.message);
    }

    // Step 3: Download
    console.log('Downloading video...');
    const downloadUrl = `http://localhost:5000/api/video/download/${jobId}`;
    window.location.href = downloadUrl;

  } catch (error) {
    console.error('Error:', error);
  }

Basic Implementation


import requests
import time
import json

API_URL = 'http://localhost:5000/api'

# Generate video
def generate_video(tiktok_links, title='RANKING'):
    response = requests.post(
        f'{API_URL}/video/generate',
        json={
            'tiktokLinks': tiktok_links,
            'bannerText': title,
            'rankingLabels': ['1', '2', '3', '4', '5'],
            'textColor': '#ffffff',
            'fontSize': 48
        }
    )
    return response.json()['jobId']

# Check status
def check_status(job_id):
    response = requests.get(f'{API_URL}/video/status/{job_id}')
    return response.json()

# Wait for completion
def wait_for_completion(job_id, timeout=300):
    start_time = time.time()
    while time.time() - start_time < timeout:
        status = check_status(job_id)
        print(f"Progress: {status['progress']}% - {status['message']}")
        
        if status['status'] == 'completed':
            return status
        elif status['status'] == 'error':
            raise Exception(f"Video generation failed: {status['message']}")
        
        time.sleep(1)
    
    raise Exception('Timeout waiting for video generation')

# Download video
def download_video(job_id, output_file):
    response = requests.get(
        f'{API_URL}/video/download/{job_id}',
        stream=True
    )
    with open(output_file, 'wb') as f:
        for chunk in response.iter_content(chunk_size=8192):
            f.write(chunk)
    print(f'Video saved to {output_file}')

# Main workflow
if __name__ == '__main__':
    tiktok_links = [
        'https://www.tiktok.com/video/...',
        'https://www.tiktok.com/video/...',
        'https://www.tiktok.com/video/...',
        'https://www.tiktok.com/video/...',
        'https://www.tiktok.com/video/...'
    ]
    
    print('Generating video...')
    job_id = generate_video(tiktok_links, 'TOP 5 CATS')
    print(f'Job ID: {job_id}')
    
    print('Waiting for completion...')
    result = wait_for_completion(job_id)
    
    print('Downloading video...')
    download_video(job_id, 'ranking-video.mp4')
    
    print('Done!')

Case 2: Batch Processing


// Process multiple video sets
const processBatch = async (videoSets) => {
  const jobs = [];

  // Start all jobs
  for (const set of videoSets) {
    const response = await fetch('http://localhost:5000/api/video/generate', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(set)
    });
    const data = await response.json();
    jobs.push(data.jobId);
  }

  // Monitor all jobs
  const results = await Promise.all(
    jobs.map(jobId => waitForCompletion(jobId))
  );

  return results;
};

// Usage
const videoSets = [
  { tiktokLinks: [...], bannerText: 'Set 1' },
  { tiktokLinks: [...], bannerText: 'Set 2' },
  { tiktokLinks: [...], bannerText: 'Set 3' }
];

const results = await processBatch(videoSets);

5️⃣ Common Patterns & Tips

Error Handling


const generateWithErrorHandling = async (links) => {
  try {
    // Validate input
    if (!Array.isArray(links) || links.length !== 5) {
      throw new Error('Exactly 5 links required');
    }

    // Generate video
    const response = await fetch('http://localhost:5000/api/video/generate', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ tiktokLinks: links })
    });

    if (!response.ok) {
      throw new Error(`API error: ${response.status}`);
    }

    const { jobId } = await response.json();
    return jobId;

  } catch (error) {
    if (error.message.includes('5 links')) {
      console.error('Invalid input:', error.message);
    } else if (error.message.includes('API error')) {
      console.error('Server error:', error.message);
    } else {
      console.error('Unexpected error:', error);
    }
    throw error;
  }
};

Retry Logic


const retryFetch = async (url, options, maxRetries = 3) => {
  for (let i = 0; i < maxRetries; i++) {
    try {
      const response = await fetch(url, options);
      if (response.ok) return response;
    } catch (error) {
      if (i === maxRetries - 1) throw error;
      await new Promise(resolve => setTimeout(resolve, 1000 * (i + 1)));
    }
  }
};

// Usage
const response = await retryFetch(
  'http://localhost:5000/api/video/generate',
  {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ tiktokLinks: [...] })
  }
);

Timeout Handling


const fetchWithTimeout = (url, options, timeout = 5000) => {
  return Promise.race([
    fetch(url, options),
    new Promise((_, reject) =>
      setTimeout(() => reject(new Error('Timeout')), timeout)
    )
  ]);
};

const generateWithTimeout = async (links) => {
  try {
    const response = await fetchWithTimeout(
      'http://localhost:5000/api/video/generate',
      {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ tiktokLinks: links })
      },
      5000
    );
    return response.json();
  } catch (error) {
    if (error.message === 'Timeout') {
      console.error('Request timeout');
    } else {
      console.error('Request failed:', error);
    }
  }
};
